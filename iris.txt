import seaborn as sns
import matplotlib.pyplot as plt

iris = sns.load_dataset('iris')

print(iris.dtypes)

iris.hist(figsize=(10,8))

plt.show()

plt.figure(figsize=(12,8))

for i, column in enumerate(iris.columns[:-1], 1):
    plt.subplot(2,2,i)
    sns.boxplot(y=iris[column])

plt.tight_layout()
plt.show()
